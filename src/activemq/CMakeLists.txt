find_package(ActiveMQcpp 3.2.0 REQUIRED)
find_package(Boost COMPONENTS filesystem thread system)
find_package(APR)
find_package(JSONCPP)

include_directories(
	${ACTIVEMQCPP_INCLUDE_DIR}
	${APR_INCLUDE_DIR}
	${JSONCPP_INCLUDE_DIRS}
)

# Disable "-Wdeprecated-declarations" warning generated by ActiveMQcpp library
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -Wno-deprecated-declarations")

# libfts_msg_ifce
set(fts_msg_ifce_SOURCES
    msg-ifce.cpp
)

add_library(fts_msg_ifce SHARED ${fts_msg_ifce_SOURCES})
target_link_libraries(fts_msg_ifce
    fts_common
    fts_msg_bus
	JSONCPP::JSONCPP
)
set_target_properties(fts_msg_ifce PROPERTIES
    LIBRARY_OUTPUT_DIRECTORY ${PROJECT_BINARY_DIR}/src/monitoring
    VERSION ${VERSION_MAJOR}.${VERSION_MINOR}.${VERSION_PATCH}
    SOVERSION ${VERSION_MAJOR}
    CLEAN_DIRECT_OUTPUT 1
)

# fts_msg_bulk binary
set(fts_msg_bulk_SOURCES
    msg-bulk-send.cpp
    MsgPipe.cpp
    MsgProducer.cpp
    BrokerConfig.cpp
)

add_executable(fts_msg_bulk ${fts_msg_bulk_SOURCES})
target_link_libraries(fts_msg_bulk
    ${ACTIVEMQCPP_LIBRARIES}
    fts_common
    fts_config
	fts_msg_bus
    rt
    ${Boost_LIBRARIES}
	JSONCPP::JSONCPP
)

# Artifacts
install(TARGETS fts_msg_bulk
	RUNTIME DESTINATION ${SBIN_INSTALL_DIR}
)

install(TARGETS fts_msg_ifce
	RUNTIME DESTINATION ${CMAKE_INSTALL_PREFIX}
	LIBRARY DESTINATION ${LIB_INSTALL_DIR}
)
